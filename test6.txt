square(int v) {
    return (v*v);
}

fact(int n) {
    int f;
    if (n == 1) {
        f = 1;
    } else {
        f = n * fact(n-1);
    }
    return f;
}

is_even(int v) {
    int ep;
    printf("%d", v);
    if (v % 2 == 0) {
        puts(" is even");
        ep = 1;
    } else {
        puts(" is odd");
        ep = 0;
    }
    return ep;
}

main() {
    printf("%d", square(7));

}
